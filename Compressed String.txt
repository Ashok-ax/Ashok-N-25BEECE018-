#include <stdio.h>
#include <ctype.h>
#include <string.h>
void expandString(const char *str) {
    int i = 0;
    while (str[i] != '\0') {
        char currentChar = str[i];
        if (str[i + 1] != '\0' && isdigit(str[i + 1])) {
            int count = str[i + 1] - '0'; 
            for (int j = 0; j < count; j++) {
                putchar(currentChar);
            }
            i += 2; 
        } else {
            putchar(currentChar);
            i++; 
        }
    }
    putchar('\n');
}
int main() {
    char input[100]; 
    if (fgets(input, sizeof(input), stdin) != NULL) 
    {
      size_t len = strlen(input);
        if (len > 0 && input[len - 1] == '\n') {
            input[len - 1] = '\0';
        }
        expandString(input);
    }
    return 0;
}
