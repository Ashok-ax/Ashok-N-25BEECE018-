#include <stdio.h>
int main() {
    int n, m;
    scanf("%d %d", &n, &m);

    int a[n][m];
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            scanf("%d", &a[i][j]);
        }
    }
    for (int r = 0; r < n; r++) {
        for (int c = 0; c < m; c++) {
            int sum = 0;
            for (int i = r - 1; i <= r + 1; i++) {
                for (int j = c - 1; j <= c + 1; j++) {
                    if (i == r && j == c)
                        continue;
                    if (i >= 0 && i < n && j >= 0 && j < m) {
                        sum += a[i][j];
                    }
                }
            }
            printf("%4d", sum);
        }
        printf("\n");
    }
    return 0;
}
