#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main(){
    vector<string> v;
    v.push_back("alpha");
    v.push_back("beta");
    v.push_back("gamma");
    cout<<"Vector: ";
    for(int i=0;i<v.size();i++) cout<<v[i]<<" ";
    cout<<endl;
    v.pop_back();
    cout<<"After pop: ";
    for(int i=0;i<v.size();i++) cout<<v[i]<<" ";
    cout<<endl;
    return 0;
}
