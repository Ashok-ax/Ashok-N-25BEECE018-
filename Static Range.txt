#include <iostream>
#include <vector>
using namespace std;
int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int n, q;
    if (!(cin >> n >> q)) return 0;
    vector<long long> prefix_sums(n + 1, 0);
    for (int i = 1; i <= n; ++i) {
        long long val;
        cin >> val;
        prefix_sums[i] = prefix_sums[i - 1] + val;
    }
    while (q--) {
        int a, b;
        cin >> a >> b;
        cout << prefix_sums[b] - prefix_sums[a - 1] << "\n";
    }

    return 0;
}
